// --- CONFIGURAÇÕES DO LAMONIÊ FLOW --- const audio = document.getElementById('audio-player'); const btnPlayPause = document.getElementById('main-play-pause'); // 1. ARRUMAR O BOTÃO PLAY/PAUSE function alternarReproducao() { if (audio.paused) { audio.play(); btnPlayPause.innerHTML = "PAUSE"; } else { audio.pause(); btnPlayPause.innerHTML = "PLAY"; } } // 2. MOSTRAR LETRA, ANO E DIA (TUDO JUNTO) function mostrarInfoCompleta(musica) { const tela = document.getElementById('detalhes-musica'); tela.style.display = 'block'; tela.innerHTML = `

${musica.titulo}

📅 Data: ${musica.data}

✨ Ano: ${musica.ano}

${musica.letra}
`; } function fechar() { document.getElementById('detalhes-musica').style.display = 'none'; } // Liga o botão ao clique no celular if(btnPlayPause) { btnPlayPause.onclick = alternarReproducao; }
